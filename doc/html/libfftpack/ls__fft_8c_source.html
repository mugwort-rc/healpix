44 complex_plan make_complex_plan (size_t length)
45  {
46  complex_plan plan = RALLOC(complex_plan_i,1);
47  size_t pfsum = prime_factor_sum(length);
48  double comp1 = (double)(length*pfsum);
49  double comp2 = 2*3*length*log(3.*length);
50  comp2*=3.; /* fudge factor that appears to give good overall performance */
51  plan->length=length;
52  plan->bluestein = (comp2<comp1);
53  if (plan->bluestein)
54  bluestein_i (length,&(plan->work),&(plan->worksize));
55  else
56  {
57  plan->worksize=4*length+15;
58  plan->work=RALLOC(double,4*length+15);
59  cffti(length, plan->work);
60  }
61  return plan;
62  }

143 void real_plan_forward_fftpack (real_plan plan, double *data)
144  {
145  if (plan->bluestein)
146  {
147  size_t m;
148  size_t n=plan->length;
149  double *tmp = RALLOC(double,2*n);
150  for (m=0; m<n; ++m)
151  {
152  tmp[2*m] = data[m];
153  tmp[2*m+1] = 0.;
154  }
155  bluestein(n,tmp,plan->work,-1);
156  data[0] = tmp[0];
157  memcpy (data+1, tmp+2, (n-1)*sizeof(double));
158  DEALLOC(tmp);
159  }
160  else
161  rfftf (plan->length, data, plan->work);
162  }

202 void real_plan_backward_fftpack (real_plan plan, double *data)
203  {
204  if (plan->bluestein)
205  {
206  size_t m;
207  size_t n=plan->length;
208  double *tmp = RALLOC(double,2*n);
209  tmp[0]=data[0];
210  tmp[1]=0.;
211  memcpy (tmp+2,data+1, (n-1)*sizeof(double));
212  if ((n&1)==0) tmp[n+1]=0.;
213  for (m=2; m<n; m+=2)
214  {
215  tmp[2*n-m]=tmp[m];
216  tmp[2*n-m+1]=-tmp[m+1];
217  }
218  bluestein (n, tmp, plan->work, 1);
219  for (m=0; m<n; ++m)
220  data[m] = tmp[2*m];
221  DEALLOC(tmp);
222  }
223  else
224  rfftb (plan->length, data, plan->work);

233 void real_plan_forward_c (real_plan plan, double *data)
234  {
235  size_t m;
236  size_t n=plan->length;
237 
238  if (plan->bluestein)
239  {
240  for (m=1; m<2*n; m+=2)
241  data[m]=0;
242  bluestein (plan->length, data, plan->work, -1);
243  data[1]=0;
244  for (m=2; m<n; m+=2)
245  {
246  double avg;
247  avg = 0.5*(data[2*n-m]+data[m]);
248  data[2*n-m] = data[m] = avg;
249  avg = 0.5*(data[2*n-m+1]-data[m+1]);
250  data[2*n-m+1] = avg;
251  data[m+1] = -avg;
252  }
253  if ((n&1)==0) data[n+1] = 0.;
254  }
255  else
256  {
257 /* using "m+m" instead of "2*m" to avoid a nasty bug in Intel's compiler */
258  for (m=0; m<n; ++m) data[m+1] = data[m+m];
259  rfftf (n, data+1, plan->work);
260  data[0] = data[1];
261  data[1] = 0;
262  for (m=2; m<n; m+=2)
263  {
264  data[2*n-m] = data[m];
265  data[2*n-m+1] = -data[m+1];
266  }
267  if ((n&1)==0) data[n+1] = 0.;
268  }
269  }

284  avg = 0.5*(data[2*n-m+1]-data[m+1]);
285  data[2*n-m+1] = avg;
286  data[m+1] = -avg;
287  }
288  if ((n&1)==0) data[n+1] = 0.;
289  bluestein (plan->length, data, plan->work, 1);
290  for (m=1; m<2*n; m+=2)
291  data[m]=0;
292  }
293  else
294  {
295  ptrdiff_t m;
296  data[1] = data[0];
297  rfftb (n, data+1, plan->work);
298  for (m=n-1; m>=0; --m)
299  {
300  data[2*m] = data[m+1];
301  data[2*m+1] = 0.;
302  }
